	$$P_{\Q} = \forall \pa{p \in \Z, q \in {\Z \setminus \set{0} }} \ba{ \frac{p}{q} \in \Q }$$
\end{definition}

	$$P^{\times} = \forall x, y \in \Z \ba { xy \in \Z }$$
